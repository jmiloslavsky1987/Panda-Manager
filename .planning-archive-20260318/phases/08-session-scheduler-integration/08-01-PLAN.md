Python algorithm to port (google_calendar.py lines 59-143):
The find_available_slots logic in JS terms:
1. Iterate freeBusyByCalendar: collect errors[], collect all busy {start,end} pairs
2. Parse ISO strings to Date objects (use new Date(str))
3. _merge_busy: sort by start, merge overlapping intervals
4. For each weekday between timeMin and timeMax (skip day.getUTCDay() === 0 or 6):
   - Compute dayStartUtc and dayEndUtc = 8am and 6pm in the user's timezone
   - Clip merged_busy to this day's window
   - Compute free periods (complement of busy in the window)
   - Walk each free period: emit slots at slotIncrement intervals while slot fits
   - Slot shape: { start: utcIso, end: utcIso, start_local: localIso, end_local: localIso, date: 'YYYY-MM-DD' }
Returns: { slots: [], errors: {} }

Timezone-to-UTC approach (Node.js, no external packages):
```javascript
// Get UTC timestamp for a specific local clock time on a specific day
function localHourToUtc(dateStr, hour, tz) {
  // 1. Create a probe Date at that hour in UTC
  const probe = new Date(`${dateStr}T${String(hour).padStart(2, '0')}:00:00Z`);
  // 2. Format it in target tz to see what local hour it actually is
  const parts = new Intl.DateTimeFormat('en-US', {
    timeZone: tz, hour: '2-digit', minute: '2-digit', hour12: false,
  }).formatToParts(probe);
  const localHour = parseInt(parts.find(p => p.type === 'hour').value);
  // 3. Adjust: if local hour is 13 but we want 8, shift back by 5h
  const adjustMs = (hour - (localHour === 24 ? 0 : localHour)) * 3600000;
  return new Date(probe.getTime() + adjustMs);
}
// For start_local and end_local, format the UTC Date back to the local timezone:
function utcToLocalIso(date, tz) {
  return date.toLocaleString('sv-SE', { timeZone: tz }).replace(' ', 'T');
  // Returns "YYYY-MM-DDTHH:MM:SS"
}
// For day-of-week check using date string (avoids local machine timezone):
function getUtcDayOfWeek(dateStr) {
  return new Date(`${dateStr}T12:00:00Z`).getUTCDay(); // 0=Sun, 6=Sat
}
```

    const calendarService = require('../services/calendarService');

    const TOKEN_PATH = path.join(__dirname, '..', 'calendar-token.json');

    function requireCalendarAuth(req, res, next) {
      try {
        if (!fs.existsSync(TOKEN_PATH)) {
          return res.status(401).json({ error: 'calendar_not_authorized' });
        }
        const raw = fs.readFileSync(TOKEN_PATH, 'utf8');
        const token = JSON.parse(raw);
        if (!token || !token.access_token) {
          return res.status(401).json({ error: 'calendar_not_authorized' });
        }
        req.calendarToken = token;
        next();
      } catch {
        return res.status(401).json({ error: 'calendar_not_authorized' });
      }
    }
