<!-- Drizzle query pattern from existing queries.ts -->
```typescript
import { db } from '@/db';
import { risks, projects } from '@/db/schema';
import { eq, ne, and, sql } from 'drizzle-orm';

// Cross-project aggregate pattern
const result = await db
  .select({ severity: risks.severity, count: sql<number>`count(*)::int` })
  .from(risks)
  .innerJoin(projects, eq(risks.project_id, projects.id))
  .where(and(eq(projects.status, 'active'), ne(risks.status, 'resolved')))
  .groupBy(risks.severity);
```
