    ```typescript
    import ExcelJS from 'exceljs';

    async function importXlsx() {
      // Find the xlsx file — handle ' copy' in filename
      const files = fs.readdirSync(SOURCE_DIR);
      const xlsxFile = files.find(f => f.includes('PA3_Action_Tracker') && f.endsWith('.xlsx'));
      if (!xlsxFile) {
        console.warn('[migrate-xlsx] PA3_Action_Tracker.xlsx not found in SOURCE_DIR — skipping');
        return;
      }
      const xlsxPath = path.join(SOURCE_DIR, xlsxFile);

      const workbook = new ExcelJS.Workbook();
      await workbook.xlsx.readFile(xlsxPath);

      // Helper: build header map from row 2
      function buildHeaderMap(sheet: ExcelJS.Worksheet): Record<string, number> {
        const headerRow = sheet.getRow(2);
        const map: Record<string, number> = {};
        headerRow.eachCell((cell, colNumber) => {
          map[String(cell.value ?? '').trim()] = colNumber;
        });
        return map;
      }

      // Helper: get project_id by customer name (normalized)
      async function getProjectId(customer: string): Promise<number | null> {
        const norm = customer.trim().toUpperCase();
        const result = await db.execute(sql`SELECT id FROM projects WHERE UPPER(customer) = ${norm} LIMIT 1`);
        return result.rows[0]?.id ?? null;
      }

      // Sheet 1: Open Actions
      const openActionsSheet = workbook.getWorksheet('Open Actions');
      if (openActionsSheet) {
        const headers = buildHeaderMap(openActionsSheet);
        let inserted = 0, skipped = 0;
        openActionsSheet.eachRow((row, rowIndex) => {
          if (rowIndex <= 2) return; // skip title and header rows
          // Use await inside a helper or collect rows first, then process
        });
        // Process rows as an array:
        const rows: ExcelJS.Row[] = [];
        openActionsSheet.eachRow((row, idx) => { if (idx > 2) rows.push(row); });
        for (const row of rows) {
          const customer = String(row.getCell(headers['Customer'])?.value ?? '');
          const externalId = String(row.getCell(headers['ID'])?.value ?? '');
          if (!externalId || !customer) continue;
          const project_id = await getProjectId(customer);
          if (!project_id) { console.warn(`[migrate-xlsx] No project found for customer '${customer}'`); continue; }
          // Idempotency: check by external_id
          const exists = await db.execute(sql`SELECT 1 FROM actions WHERE external_id = ${externalId} AND project_id = ${project_id} LIMIT 1`);
          if (exists.rows.length > 0) { skipped++; continue; }
          await db.insert(actions).values({
            project_id,
            external_id: externalId,
            description: String(row.getCell(headers['Description'])?.value ?? ''),
            owner: String(row.getCell(headers['Owner'])?.value ?? '') || null,
            due: String(row.getCell(headers['Due'])?.value ?? '') || null,
            status: 'open',
            last_updated: String(row.getCell(headers['Last Updated'])?.value ?? '') || null,
            notes: String(row.getCell(headers['Notes'])?.value ?? '') || null,
            source: 'xlsx_open',
          });
          inserted++;
        }
        console.log(`[migrate-xlsx] Open Actions: ${inserted} inserted, ${skipped} skipped`);
      }

      // Sheet 2: Open Risks — same pattern, source='xlsx_risks', check by (project_id, external_id)
      // Sheet 3: Open Questions — same as actions but type='question', source='xlsx_questions'
      // Sheet 4: Workstream Notes — UPDATE workstreams SET current_status=..., lead=... WHERE project_id AND name=...
      // Sheet 5: Completed — same as Open Actions but status='completed', source='xlsx_completed'
    }
    ```
