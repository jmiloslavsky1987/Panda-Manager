    ```typescript
    'use client';
    import CodeMirror from '@uiw/react-codemirror';
    import { markdown } from '@codemirror/lang-markdown';
    import { oneDark } from '@codemirror/theme-one-dark';

    interface CodeMirrorEditorProps {
      defaultValue: string;
      onChange: (value: string) => void;
      isFullScreen?: boolean;
    }

    export default function CodeMirrorEditor({ defaultValue, onChange, isFullScreen }: CodeMirrorEditorProps) {
      return (
        <CodeMirror
          value={defaultValue}
          height={isFullScreen ? '70vh' : '400px'}
          extensions={[markdown()]}
          theme={oneDark}
          onChange={onChange}
          basicSetup={{ lineNumbers: true, foldGutter: false }}
        />
      );
    }
    ```

    CRITICAL: Use `value` prop (not `defaultValue`) with `@uiw/react-codemirror` — the library handles change tracking internally. Use a `ref` to buffer the current value rather than React state to avoid cursor-jump pitfall. The `onChange` callback is called with the full current document on every change — store in a `useRef` in the parent (PromptEditModal), not useState.

    Actually: Per pitfall warning in RESEARCH.md — use `defaultValue` on the CodeMirror component (not `value`) to avoid controlled-mode cursor jumps. The component name prop is `defaultValue` but internally pass it as the `value` for initial load only. Use `onChange` to buffer in a ref in the parent. This is uncontrolled mode.
