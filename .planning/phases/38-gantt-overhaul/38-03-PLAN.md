<!-- frappe-gantt SVG structure (for marker positioning):
  frappe-gantt renders an <svg> with internal <g class="dates"> containing date column positions.
  The timeline x-axis is NOT easily queryable by date — frappe-gantt does not expose a public
  API to convert a date to an x-coordinate.

  APPROACH: Use a CSS-based overlay approach. The marker is positioned using the milestone date
  relative to the visible date range of the Gantt chart, calculated from the tasks' start/end dates.

  Calculate the chart's date range from the task data:
  - chartStart = min(task.start) across all tasks in this lane
  - chartEnd = max(task.end) across all tasks in this lane
  - markerLeft = (markerDate - chartStart) / (chartEnd - chartStart) * 100%

  The marker is a position:absolute div overlaid on the svg container, pointer-events:none
  except for the label (which has pointer-events:auto for click popup).

  The SVG container wrapper needs position:relative to anchor the overlays.
-->

Create bigpanda-app/components/MilestoneMarker.tsx:

```tsx
'use client'
import { useState } from 'react'

interface MilestoneMarkerProps {
  name: string
  date: string         // ISO 'YYYY-MM-DD'
  status: string | null
  leftPercent: number  // 0-100, horizontal position within chart container
  staggerRow: 0 | 1   // 0 = top label row, 1 = second label row (stagger)
}

export function MilestoneMarker({ name, date, status, leftPercent, staggerRow }: MilestoneMarkerProps) {
  const [open, setOpen] = useState(false)
  const topOffset = staggerRow === 0 ? 4 : 20

  return (
    <div
      className="absolute top-0 bottom-0 z-10"
      style={{ left: `${leftPercent}%`, transform: 'translateX(-50%)' }}
    >
      {/* Dashed vertical line spanning full chart height */}
      <div
        className="absolute top-0 bottom-0 w-px border-l-2 border-dashed border-indigo-400 pointer-events-none"
        aria-hidden="true"
      />

      {/* Milestone name label — clickable */}
      <button
        className="absolute left-1 text-xs text-indigo-600 font-medium whitespace-nowrap bg-white/80 px-1 rounded leading-tight hover:text-indigo-800 transition-colors"
        style={{ top: topOffset }}
        onClick={() => setOpen(o => !o)}
        aria-label={`Milestone: ${name}`}
      >
        {name}
      </button>

      {/* Click popup */}
      {open && (
        <>
          {/* Click-away backdrop */}
          <div
            className="fixed inset-0 z-20"
            onClick={() => setOpen(false)}
            aria-hidden="true"
          />
          <div className="absolute left-3 z-30 bg-white border border-zinc-200 rounded-md shadow-lg p-3 min-w-max text-xs"
               style={{ top: topOffset + 18 }}>
            <p className="font-semibold text-zinc-800 mb-1">{name}</p>
            <p className="text-zinc-500">Target: {date}</p>
            {status && <p className="text-zinc-500 mt-0.5">Status: <span className="capitalize">{status.replace(/_/g, ' ')}</span></p>}
          </div>
        </>
      )}
    </div>
  )
}
```

2. Add `milestones: GanttMilestone[]` prop to SwimLaneChartProps (all milestones with valid dates are overlaid on every lane — the user sees which milestones fall in the timeline regardless of which lane they're on)
3. Calculate leftPercent for each milestone marker based on date range:

```typescript
// Helper: days between two ISO date strings
function daysBetween(a: string, b: string): number {
  return (new Date(b).getTime() - new Date(a).getTime()) / 86_400_000
}

// In SwimLaneChart render:
const markerableMilestones = milestones.filter(m => m.date !== null) as (GanttMilestone & { date: string })[]

// Calculate chart date range from tasks
const taskDates = tasks.flatMap(t => [t.start, t.end]).sort()
const chartStart = taskDates[0]
const chartEnd = taskDates[taskDates.length - 1]
const totalDays = daysBetween(chartStart, chartEnd)

// Calculate left positions, filter out-of-range markers
const markersInRange = markerableMilestones
  .filter(m => m.date >= chartStart && m.date <= chartEnd)
  .sort((a, b) => a.date.localeCompare(b.date))
  .map((m, i, arr) => {
    const leftPercent = totalDays > 0
      ? (daysBetween(chartStart, m.date) / totalDays) * 100
      : 50
    // Stagger: check if previous marker is within 14 days
    const prev = arr[i - 1]
    const closeToPrev = prev && daysBetween(prev.date, m.date) < 14
    const staggerRow: 0 | 1 = closeToPrev ? 1 : 0
    return { ...m, leftPercent, staggerRow }
  })
```

This is acceptable per scope — exact pixel-perfect alignment is not required. The markers will be approximately correct and visually useful.
