Pattern for PATCH (update) in production code:
```typescript
const [before] = await db.select().from(artifacts).where(eq(artifacts.id, numericId))
if (!before) return NextResponse.json({ error: 'Not found' }, { status: 404 })

await db.transaction(async (tx) => {
  await tx.update(artifacts).set(patch).where(eq(artifacts.id, numericId))
  await tx.insert(auditLog).values({
    entity_type: 'artifact',
    entity_id: numericId,
    action: 'update',
    actor_id: 'default',
    before_json: before as Record<string, unknown>,
    after_json: { ...before, ...patch } as Record<string, unknown>,
  })
})
```

Pattern for CREATE with .returning():
```typescript
await db.transaction(async (tx) => {
  const [inserted] = await tx.insert(table).values({...}).returning()
  await tx.insert(auditLog).values({
    entity_type: 'task',
    entity_id: inserted.id,
    action: 'create',
    actor_id: 'default',
    before_json: null,
    after_json: inserted as Record<string, unknown>,
  })
})
```

From bigpanda-app/db/schema.ts (auditLog table):
```typescript
export const auditLog = pgTable('audit_log', {
  id:          serial('id').primaryKey(),
  entity_type: text('entity_type').notNull(),
  entity_id:   integer('entity_id'),
  action:      text('action').notNull(),  // 'create' | 'update' | 'delete'
  actor_id:    text('actor_id'),
  before_json: jsonb('before_json'),
  after_json:  jsonb('after_json'),
  created_at:  timestamp('created_at').defaultNow(),
})
```

NOTE on writeAuditLog helper: It does NOT accept a transaction `tx` parameter — it uses the top-level `db`. For transactional atomicity, use inline `tx.insert(auditLog).values({...})` inside `db.transaction()` as shown in the reference above.

IMPORTANT PITFALL — findConflict() returns only { id: number }:
The `findConflict()` function in ingestion/approve/route.ts selects ONLY `{ id }` from each table. For merge before_json, do NOT use the findConflict result directly as before_json — it only contains { id: N }.
Solution: After findConflict returns a hit, do a full-record SELECT on the specific table using the returned id to get before_json. Example:
```typescript
case 'merge': {
  const idToMerge = item.existingId ?? existing.id;
  // Fetch full before-state for audit (findConflict only returns { id })
  const [beforeRecord] = await db.select().from(actions).where(eq(actions.id, idToMerge))
  await db.transaction(async (tx) => {
    await tx.update(actions).set({...}).where(eq(actions.id, idToMerge))
    const [afterRecord] = await tx.select().from(actions).where(eq(actions.id, idToMerge))
    await tx.insert(auditLog).values({
      entity_type: 'action',
      entity_id: idToMerge,
      action: 'update',
      actor_id: 'default',
      before_json: beforeRecord as Record<string, unknown>,
      after_json: afterRecord as Record<string, unknown>,
    })
  })
  break;
}
```
