Create `bigpanda-app/app/api/actions/[id]/route.ts`.

**Critical constraint (from CONTEXT.md):** xlsx write FIRST, then DB write. If xlsx write fails, return error and do NOT write DB. Never silently diverge xlsx and DB.

**Import pattern:**
```typescript
import { NextRequest } from 'next/server'
import { z } from 'zod'
import ExcelJS from 'exceljs'
import * as path from 'path'
import { db } from '@/db'
import { actions } from '@/db/schema'
import { eq } from 'drizzle-orm'
import { readSettings } from '@/lib/settings'
```

**Input validation with Zod:**
```typescript
const ActionPatchSchema = z.object({
  description: z.string().min(1).optional(),
  owner: z.string().optional(),
  due: z.string().optional(),
  status: z.enum(['open', 'in_progress', 'completed', 'cancelled']).optional(),
  notes: z.string().optional(),
})
type ActionPatch = z.infer<typeof ActionPatchSchema>
```

**PATCH handler:**
```typescript
export async function PATCH(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }   // Next.js 15 async params
) {
  const { id } = await params
  const actionId = parseInt(id)
  if (isNaN(actionId)) return Response.json({ error: 'Invalid action ID' }, { status: 400 })

  let patch: ActionPatch
  try {
    patch = ActionPatchSchema.parse(await req.json())
  } catch (err) {
    return Response.json({ error: 'Invalid request body' }, { status: 400 })
  }

  try {
    // Step 1: xlsx write FIRST
    await updateXlsxRow(actionId, patch)

    // Step 2: DB write (only after xlsx succeeds)
    await db.update(actions)
      .set({ ...patch, last_updated: new Date().toISOString().split('T')[0] })
      .where(eq(actions.id, actionId))

    return Response.json({ ok: true })
  } catch (err) {
    const message = err instanceof Error ? err.message : 'Write failed'
    return Response.json({ error: message }, { status: 500 })
  }
}
```

**updateXlsxRow function:**
- Read settings to get workspace_path
- Build xlsxPath = `path.join(settings.workspace_path, 'PA3_Action_Tracker.xlsx')`
- If file does not exist: skip xlsx update silently (no error — xlsx may not be set up yet in dev) and return
- Load workbook: `const wb = new ExcelJS.Workbook(); await wb.xlsx.readFile(xlsxPath)`
- Get action's external_id from DB: `const [action] = await db.select({ external_id: actions.external_id }).from(actions).where(eq(actions.id, actionId)).limit(1)`
- Get Open Actions sheet: try `wb.getWorksheet('🔴 Open Actions')` then `wb.getWorksheet('Open Actions')` as fallback
- Build header map from row 2 (same pattern as migrate-local.ts)
- Find row matching external_id (rows starting at index 3)
- If `patch.status === 'completed'`: move row to `✅ Completed` sheet (or `Completed` sheet as fallback):
  - Copy all cell values to a new row appended to Completed sheet
  - Delete the row from Open Actions sheet (call `sheet.spliceRows(rowIndex, 1)`)
- Otherwise: update cells in-place for changed fields only
- Write file: `await wb.xlsx.writeFile(xlsxPath)` — this throws if file is locked (EBUSY/EPERM)
- On EBUSY/EPERM: throw `new Error('Close PA3_Action_Tracker.xlsx in Excel before saving')`

**Note:** Use `new ExcelJS.Workbook()` (document-based, NOT streaming) to preserve cell styles and themes.

```tsx
// Still RSC — no 'use client' on this file
import { getWorkspaceData } from '@/lib/queries'
import { ActionEditModal } from '@/components/ActionEditModal'

export default async function ActionsPage({
  params,
}: {
  params: Promise<{ id: string }>
}) {
  const { id } = await params
  const projectId = parseInt(id)
  const data = await getWorkspaceData(projectId)

  return (
    <div data-testid="actions-tab" className="p-4">
      <h2 className="text-lg font-semibold mb-4">Actions</h2>
      <div className="space-y-2">
        {data.actions.map((action) => (
          <ActionEditModal
            key={action.id}
            action={action}
            trigger={
              <div className="border rounded p-3 cursor-pointer hover:bg-zinc-50">
                <div className="flex items-center justify-between">
                  <span className="font-mono text-xs text-zinc-400">{action.external_id}</span>
                  <span className={`text-xs px-2 py-0.5 rounded-full ${
                    action.status === 'completed' ? 'bg-green-100 text-green-700' :
                    action.status === 'in_progress' ? 'bg-blue-100 text-blue-700' :
                    action.status === 'cancelled' ? 'bg-zinc-100 text-zinc-500' :
                    'bg-yellow-100 text-yellow-700'
                  }`}>{action.status}</span>
                </div>
                <p className="text-sm mt-1 text-zinc-800">{action.description}</p>
                <div className="flex gap-4 mt-1 text-xs text-zinc-500">
                  <span>Owner: {action.owner ?? '—'}</span>
                  <span>Due: {action.due ?? '—'}</span>
                </div>
              </div>
            }
          />
        ))}
        {data.actions.length === 0 && (
          <p className="text-zinc-500 text-sm">No actions found.</p>
        )}
      </div>
    </div>
  )
}
```
