<!-- PATCH helpers (mirrors Plan 04 pattern) -->
async function patchRisk(id: number, patch: Record<string, unknown>, router) {
  const res = await fetch(`/api/risks/${id}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(patch),
  })
  if (!res.ok) throw new Error('Save failed')
  router.refresh()
}

async function patchMilestone(id: number, patch: Record<string, unknown>, router) {
  const res = await fetch(`/api/milestones/${id}`, {
    method: 'PATCH',
    headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify(patch),
  })
  if (!res.ok) throw new Error('Save failed')
  router.refresh()
}
