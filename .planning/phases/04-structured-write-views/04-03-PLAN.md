Pitfall 5 — coerce percent_complete values to integers:
```javascript
// Server-side normalization before unshift
const normalizeEntry = (entry) => {
  const normalized = { ...entry };
  if (normalized.workstreams && typeof normalized.workstreams === 'object') {
    for (const [groupKey, group] of Object.entries(normalized.workstreams)) {
      for (const [subKey, sub] of Object.entries(group)) {
        normalized.workstreams = {
          ...normalized.workstreams,
          [groupKey]: {
            ...normalized.workstreams[groupKey],
            [subKey]: {
              ...sub,
              percent_complete: parseInt(sub.percent_complete, 10) || 0,
            },
          },
        };
      }
    }
  }
  return normalized;
};
```
