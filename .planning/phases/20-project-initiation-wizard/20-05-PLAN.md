    ```typescript
    import { NextRequest, NextResponse } from 'next/server'
    import { db } from '@/db'
    import { actions, risks, milestones, stakeholders, keyDecisions,
             architectureIntegrations, teamOnboardingStatus,
             engagementHistory, businessOutcomes } from '@/db/schema'
    import { eq, count } from 'drizzle-orm'

    export function computeCompletenessScore(populatedCount: number): number {
      return Math.round((populatedCount / 9) * 100)
    }

    export function getBannerData(emptyTabs: string[]): string[] {
      return emptyTabs
    }

    const SCORED_TABS = [
      { label: 'Actions',            table: actions,                path: 'actions' },
      { label: 'Risks',              table: risks,                  path: 'risks' },
      { label: 'Milestones',         table: milestones,             path: 'milestones' },
      { label: 'Stakeholders',       table: stakeholders,           path: 'stakeholders' },
      { label: 'Decisions',          table: keyDecisions,           path: 'decisions' },
      { label: 'Architecture',       table: architectureIntegrations, path: 'architecture' },
      { label: 'Teams',              table: teamOnboardingStatus,   path: 'teams' },
      { label: 'Engagement History', table: engagementHistory,      path: 'history' },
      { label: 'Business Outcomes',  table: businessOutcomes,       path: 'plan' },
    ] as const

    export async function GET(
      _req: NextRequest,
      { params }: { params: Promise<{ projectId: string }> }
    ) {
      const { projectId } = await params
      const numericId = parseInt(projectId, 10)
      if (isNaN(numericId)) {
        return NextResponse.json({ error: 'Invalid projectId' }, { status: 400 })
      }

      const results = await Promise.all(
        SCORED_TABS.map(async (tab) => {
          const rows = await db
            .select({ cnt: count() })
            .from(tab.table)
            .where(eq(tab.table.project_id, numericId))
          return { label: tab.label, path: tab.path, populated: (rows[0]?.cnt ?? 0) > 0 }
        })
      )

      const populatedTabs = results.filter(r => r.populated).map(r => r.label)
      const emptyTabs = results.filter(r => !r.populated).map(r => r.label)
      const score = computeCompletenessScore(populatedTabs.length)

      return NextResponse.json({ score, populatedTabs, emptyTabs })
    }
    ```

    ```typescript
    import { OnboardingDashboard } from '../../../../components/OnboardingDashboard'
    import Link from 'next/link'

    export default async function OverviewPage({
      params,
    }: {
      params: Promise<{ id: string }>
    }) {
      const { id } = await params
      const projectId = parseInt(id, 10)

      // Fetch completeness server-side (same-origin fetch in RSC)
      let completeness = { score: 0, emptyTabs: [] as string[] }
      try {
        const res = await fetch(`http://localhost:3000/api/projects/${id}/completeness`, {
          cache: 'no-store',
        })
        if (res.ok) {
          completeness = await res.json()
        }
      } catch {
        // Completeness unavailable — continue without it
      }

      const { score, emptyTabs } = completeness
      const showBanner = score < 60 && emptyTabs.length > 0

      return (
        <div className="space-y-4">
          {/* Completeness Score Bar — always visible (WIZ-08, WIZ-09) */}
          <div className="px-6 pt-4">
            <div className="flex items-center justify-between mb-1">
              <span className="text-sm font-medium text-zinc-700">Project Completeness</span>
              <span className="text-sm font-semibold text-zinc-900">{score}%</span>
            </div>
            <div className="w-full bg-zinc-200 rounded-full h-2">
              <div
                className="bg-blue-500 h-2 rounded-full transition-all"
                style={{ width: `${score}%` }}
              />
            </div>
          </div>

          {/* Below-60% warning banner — non-dismissible, disappears when score >= 60% */}
          {showBanner && (
            <div className="mx-6 rounded border border-yellow-400 bg-yellow-50 px-4 py-3">
              <p className="text-sm font-medium text-yellow-800 mb-1">
                Missing data in:{' '}
                {emptyTabs.map((tab, i) => (
                  <span key={tab}>
                    <Link
                      href={`/customer/${id}/${getTabPath(tab)}`}
                      className="underline hover:text-yellow-900"
                    >
                      {tab}
                    </Link>
                    {i < emptyTabs.length - 1 ? ', ' : ''}
                  </span>
                ))}
              </p>
              <p className="text-xs text-yellow-700">
                Add data to these tabs to reach 60% completeness.
              </p>
            </div>
          )}

          <OnboardingDashboard projectId={projectId} />
        </div>
      )
    }

    // Maps tab label to URL path segment
    function getTabPath(tabLabel: string): string {
      const map: Record<string, string> = {
        'Actions': 'actions',
        'Risks': 'risks',
        'Milestones': 'milestones',
        'Stakeholders': 'stakeholders',
        'Decisions': 'decisions',
        'Architecture': 'architecture',
        'Teams': 'teams',
        'Engagement History': 'history',
        'Business Outcomes': 'plan',
      }
      return map[tabLabel] ?? tabLabel.toLowerCase().replace(' ', '-')
    }
    ```
