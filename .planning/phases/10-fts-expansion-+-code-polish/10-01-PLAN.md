From bigpanda-app/db/migrations/0008_fts_and_kb.sql — established FTS pattern (apply for each of 4 tables):
```sql
ALTER TABLE {table} ADD COLUMN IF NOT EXISTS search_vec tsvector;
CREATE INDEX IF NOT EXISTS idx_{table}_search_vec ON {table} USING GIN(search_vec);

CREATE OR REPLACE FUNCTION tsvector_update_{table}() RETURNS trigger AS $$
BEGIN
  NEW.search_vec := to_tsvector('english', coalesce(NEW.field1, '') || ' ' || coalesce(NEW.field2, ''));
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

DROP TRIGGER IF EXISTS trg_{table}_search_vec ON {table};
CREATE TRIGGER trg_{table}_search_vec
  BEFORE INSERT OR UPDATE ON {table}
  FOR EACH ROW EXECUTE FUNCTION tsvector_update_{table}();

UPDATE {table} SET search_vec = to_tsvector('english', coalesce(field1, '') || ' ' || coalesce(field2, ''));
```
