    ) {
      const { id } = await params;
      const numericId = parseInt(id, 10);
      if (isNaN(numericId)) {
        return NextResponse.json({ error: 'Invalid id' }, { status: 400 });
      }

      const [entity] = await db.select({ project_id: {table}.project_id })
        .from({table})
        .where(eq({table}.id, numericId));
      if (!entity) return NextResponse.json({ error: 'Not found' }, { status: 404 });

      const { session, redirectResponse } = await requireProjectRole(entity.project_id, 'user');
      if (redirectResponse) return redirectResponse;

      const [before] = await db.select().from({table}).where(eq({table}.id, numericId));
      if (!before) return NextResponse.json({ error: 'Not found' }, { status: 404 });

      await db.transaction(async (tx) => {
        await tx.execute(sql.raw(`SET LOCAL app.current_project_id = ${entity.project_id}`));
        await tx.insert(auditLog).values({

          before_json: before as Record<string, unknown>,
          after_json: null,
        });
        await tx.delete({table}).where(eq({table}.id, numericId));
      });

      return NextResponse.json({ success: true });
    }
