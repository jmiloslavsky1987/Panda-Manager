**TeamEngagementMap.tsx** — 'use client' wrapper that holds state (optimistic data) and renders 5 sections. Pass `architectureIntegrations` from `data` into `ArchOverviewSection`:
```typescript
'use client'
import { useState } from 'react'
import { TeamsTabData, BusinessOutcome, E2eWorkflowWithSteps, FocusArea, ArchitectureIntegration } from '@/lib/queries'
import { BusinessOutcomesSection } from './BusinessOutcomesSection'
import { ArchOverviewSection } from './ArchOverviewSection'
import { E2eWorkflowsSection } from './E2eWorkflowsSection'
import { TeamsEngagementSection } from './TeamsEngagementSection'
import { FocusAreasSection } from './FocusAreasSection'

interface Props {
  projectId: number
  customer: string
  data: TeamsTabData
}

export function TeamEngagementMap({ projectId, customer, data }: Props) {
  const [outcomes, setOutcomes] = useState(data.businessOutcomes)
  const [workflows, setWorkflows] = useState(data.e2eWorkflows)
  const [focusAreas, setFocusAreas] = useState(data.focusAreas)

  return (
    <div className="space-y-10">
      <BusinessOutcomesSection projectId={projectId} outcomes={outcomes} onUpdate={setOutcomes} />
      <ArchOverviewSection integrations={data.architectureIntegrations} />
      <E2eWorkflowsSection projectId={projectId} workflows={workflows} onUpdate={setWorkflows} />
      <TeamsEngagementSection customer={customer} workflows={workflows} />
      <FocusAreasSection projectId={projectId} focusAreas={focusAreas} onUpdate={setFocusAreas} />
    </div>
  )
}
```

Note: `architectureIntegrations` is read-only in the Teams tab — no optimistic state needed; pass directly from `data`.
