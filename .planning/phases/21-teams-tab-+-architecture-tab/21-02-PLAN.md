```typescript
export async function getArchTabData(projectId: number): Promise<ArchTabData> {
  const [integrations, beforeStateRows, onboardingRows] = await Promise.all([
    db.select().from(architectureIntegrations)
      .where(eq(architectureIntegrations.project_id, projectId))
      .orderBy(asc(architectureIntegrations.phase), asc(architectureIntegrations.tool_name)),
    db.select().from(beforeState).where(eq(beforeState.project_id, projectId)).limit(1),
    db.select().from(teamOnboardingStatus)
      .where(eq(teamOnboardingStatus.project_id, projectId))
      .orderBy(asc(teamOnboardingStatus.team_name)),
  ]);
  return {
    architectureIntegrations: integrations,
    beforeState: beforeStateRows[0] ?? null,
    teamOnboardingStatus: onboardingRows,
  };
}
```

**architecture-integrations/route.ts:**
- GET: SELECT all from architectureIntegrations WHERE project_id, ordered by phase, tool_name. Return `{ integrations: [...] }`
- POST: INSERT (tool_name, track, phase, status, integration_method, notes). Return `{ integration: result[0] }`

**architecture-integrations/[id]/route.ts:**
- PATCH: UPDATE fields (tool_name, track, phase, status, integration_method, notes). Return `{ integration: result[0] }`
- DELETE: DELETE WHERE id + project_id. Return `{ success: true }`

**before-state/route.ts:**
- GET: SELECT one row WHERE project_id. Return `{ beforeState: row | null }`
- PUT: Upsert. If a row exists for this project_id, UPDATE it. If not, INSERT. Accept body fields: aggregation_hub_name, alert_to_ticket_problem, pain_points_json (array of strings). Return `{ beforeState: result }`
  Implementation: First try UPDATE. If rowCount === 0, INSERT. Use db.transaction for atomicity.

**team-onboarding-status/route.ts:**
- GET: SELECT all from teamOnboardingStatus WHERE project_id. Return `{ rows: [...] }`
- POST: INSERT (team_name, track, ingest_status, correlation_status, incident_intelligence_status, sn_automation_status, biggy_ai_status). Return `{ row: result[0] }`

**team-onboarding-status/[id]/route.ts:**
- PATCH: UPDATE status fields by row id. Return `{ row: result[0] }`
- DELETE: DELETE WHERE id + project_id. Return `{ success: true }`

All routes: validate projectId integer, wrap in try/catch (500 on error), use SET LOCAL RLS in db.transaction.
