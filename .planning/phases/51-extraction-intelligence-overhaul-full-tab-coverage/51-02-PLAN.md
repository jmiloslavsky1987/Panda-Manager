    // Exported status coercer functions for entity types added/fixed in Phase 51.
    // Follow the same pattern as coerceTrackStatus in approve/route.ts.

    // Gap E: wbsItemStatusEnum values: not_started, in_progress, complete (underscores, no "blocked")
    export type WbsItemStatus = 'not_started' | 'in_progress' | 'complete';
    export function coerceWbsItemStatus(raw: string | undefined | null): WbsItemStatus | null {
      if (!raw) return null;
      const v = raw.toLowerCase().trim();
      if (['complete', 'completed', 'done', 'finished'].includes(v)) return 'complete';
      if (['in_progress', 'in progress', 'in-progress', 'ongoing', 'running', 'active'].includes(v)) return 'in_progress';
      if (['not_started', 'not started', 'todo', 'pending', 'not-started'].includes(v)) return 'not_started';
      return null; // 'blocked' and other free-text → null (not in enum)
    }

    // Gap E: archNodeStatusEnum values: planned, in_progress, live (no "not_started")
    export type ArchNodeStatus = 'planned' | 'in_progress' | 'live';
    export function coerceArchNodeStatus(raw: string | undefined | null): ArchNodeStatus | null {
      if (!raw) return null;
      const v = raw.toLowerCase().trim();
      if (['live', 'production', 'prod', 'active', 'enabled', 'running'].includes(v)) return 'live';

      if (['planned', 'scheduled', 'upcoming', 'not started', 'not_started', 'not-started', 'pending'].includes(v)) return 'planned';
      return null;
    }
