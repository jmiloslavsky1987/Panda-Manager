```typescript
import type { Job } from 'bullmq';
import { eq } from 'drizzle-orm';
import db from '../../db';
import { skillRuns } from '../../db/schema';
import { readSettings } from '../../lib/settings-core';  // NOT lib/settings (server-only!)

export default async function skillRunJob(job: Job): Promise<{ status: string }> {
  const { runId, ... } = job.data;

  // 1. Mark running
  await db.update(skillRuns).set({ status: 'running', updated_at: new Date() }).where(eq(skillRuns.id, runId));

  try {
    // 2. Do work ...
    // 3. Mark completed
    await db.update(skillRuns).set({ status: 'completed' }).where(eq(skillRuns.id, runId));
    return { status: 'completed' };
  } catch (err) {
    // 4. Mark failed + re-throw
    await db.update(skillRuns).set({ status: 'failed', error_message: String(err) }).where(eq(skillRuns.id, runId));
    throw err;
  }
}
```
