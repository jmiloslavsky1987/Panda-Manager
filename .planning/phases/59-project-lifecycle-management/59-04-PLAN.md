From bigpanda-app/app/customer/[id]/members/page.tsx (isProjectAdmin resolution pattern to copy):
```typescript
import { auth } from '@/lib/auth'
import { headers } from 'next/headers'
import { db } from '@/db'
import { projectMembers } from '@/db/schema'
import { eq, and } from 'drizzle-orm'
import { resolveRole } from '@/lib/auth-utils'

const session = await auth.api.getSession({ headers: await headers() })
let isProjectAdmin = false
if (session?.user) {
  if (resolveRole(session) === 'admin') {
    isProjectAdmin = true
  } else {
    const [member] = await db.select({ role: projectMembers.role })
      .from(projectMembers)
      .where(and(eq(projectMembers.project_id, projectId), eq(projectMembers.user_id, session.user.id)))
      .limit(1)
    isProjectAdmin = member?.role === 'admin'
  }
}
```
