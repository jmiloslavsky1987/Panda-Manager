    ```tsx
    'use client'
    import { useEffect, type ReactNode } from 'react'
    import { usePageBar } from './PageBarContext'
    import { Badge } from './ui/badge'

    const RAG_CLASSES = {
      green: 'bg-green-100 text-green-800',
      yellow: 'bg-yellow-100 text-yellow-800',
      red: 'bg-red-100 text-red-800',
    }
    const RAG_LABELS = { green: 'Healthy', yellow: 'At Risk', red: 'Critical' }

    export function WorkspacePageBarConfigurator({
      title,
      health,
      ctaSlot,
    }: {
      title: string
      health: 'green' | 'yellow' | 'red'
      ctaSlot?: ReactNode
    }) {
      const { setTitle, setCtaSlot } = usePageBar()
      useEffect(() => {
        setTitle(title)
        setCtaSlot(
          <div className="flex items-center gap-3">
            <Badge className={RAG_CLASSES[health]}>{RAG_LABELS[health]}</Badge>
            {ctaSlot}
          </div>
        )
        return () => { setTitle(''); setCtaSlot(null) }
      }, [title, health])
      return null
    }
    ```
