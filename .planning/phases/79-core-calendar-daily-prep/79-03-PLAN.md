    ```typescript
    const [projects, setProjects] = useState<Project[]>([])

    const [loading, setLoading] = useState(false)
    ```

    useEffect for calendar status check (runs once on mount):
    ```typescript
    fetch('/api/oauth/calendar/status').then(r => r.json()).then(d => setConnected(d.connected))
    ```

    useEffect for event fetch (runs when selectedDate changes):
    ```typescript
    if (!connected) return
    setLoading(true)
    fetch(`/api/time-entries/calendar-import?date=${selectedDate}`)
      .then(r => r.json())
      .then((events: CalendarEventItem[]) => {
        // Load stored briefs from LocalStorage
        const storageKey = `daily-prep-briefs:${selectedDate}`
        const stored: StoredBriefs = JSON.parse(localStorage.getItem(storageKey) ?? '{}')
        setCards(events.map(event => ({
          event,
          selectedProjectId: event.matched_project_id,
          selected: false,
          briefStatus: stored[event.event_id] ? 'done' : 'idle',
          briefContent: stored[event.event_id]?.content ?? null,
          expanded: !!stored[event.event_id],
        })))
      })
      .finally(() => setLoading(false))
    ```
