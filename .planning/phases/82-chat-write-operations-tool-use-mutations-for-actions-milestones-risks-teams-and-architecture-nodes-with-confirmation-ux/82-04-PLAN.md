```typescript
import { isToolUIPart } from 'ai'

// In message parts rendering:
message.parts.map((part, partIndex) => {
  if (isToolUIPart(part)) {
    if (part.state === 'approval-requested') {
      return <MutationConfirmCard
        key={`${message.id}-${partIndex}`}
        part={part as any}
        onApprove={() => addToolApprovalResponse({ id: (part as any).approval.id, approved: true })}
        onCancel={() => addToolApprovalResponse({ id: (part as any).approval.id, approved: false, reason: 'User cancelled' })}
      />
    }
    // approval-responded, output-available, error:
    return <MutationConfirmCardComplete key={`${message.id}-${partIndex}`} part={part as any} />
  }
  // text parts:
  return <ChatMessage key={`${message.id}-${partIndex}`} message={message as any} partIndex={partIndex} />
})
```
