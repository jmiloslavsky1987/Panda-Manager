```typescript
export const createActionTool = (projectId: number) => tool({
  description: 'Create a new action item for this project',
  inputSchema: zodSchema(z.object({ /* fields above */ })),
  needsApproval: true,
  execute: async (input) => {
    // Use dynamic imports for Docker build compatibility (pattern from [80-03], [80-04])
    const { default: db } = await import('@/db')
    const { actions } = await import('@/db/schema')
    // Insert with source: 'chat', project_id from closure
    const [created] = await db.insert(actions).values({
      ...input,
      project_id: projectId,
      source: 'chat',
      external_id: `A-CHAT-${Date.now()}`, // temporary external_id
    }).returning({ id: actions.id })
    return { success: true, id: created.id }
  },
})
```
