```typescript
import Anthropic from '@anthropic-ai/sdk';
import type { MCPServerConfig } from '../settings-core';
import { SlackAdapter } from './slack-adapter';
import { GongAdapter } from './gong-adapter';
import { GleanAdapter } from './glean-adapter';
import { GmailAdapter } from './gmail-adapter';
import { MCPAdapter } from './mcp-adapter';

// (keep existing interfaces: SourceAdapter, SourceCredentials, SourceName, UserSourceToken)

const SOURCE_TOOL_MAP: Record<string, string> = {
  slack: 'search_messages',
  gmail: 'search_emails',
  glean: 'search_documents',
  gong: 'get_transcripts',
};

export function resolveAdapter(
  source: SourceName,
  orgCredentials: SourceCredentials,
  userToken: UserSourceToken | null,
  mcpServer: MCPServerConfig | undefined
): SourceAdapter | null {
  // REST credentials take priority over MCP (per user decision)

  if (source === 'slack' && orgCredentials.slack?.token) {
    return new SlackAdapter(orgCredentials.slack.token, orgCredentials.slack.channels ?? []);
  }

  if (source === 'gong' && orgCredentials.gong?.accessKey && orgCredentials.gong?.accessKeySecret && orgCredentials.gong?.baseUrl) {
    return new GongAdapter(orgCredentials.gong.accessKey, orgCredentials.gong.accessKeySecret, orgCredentials.gong.baseUrl);
  }

  if (source === 'glean' && orgCredentials.glean?.token && orgCredentials.glean?.instanceUrl) {
    return new GleanAdapter(orgCredentials.glean.token, orgCredentials.glean.instanceUrl, orgCredentials.glean.actAsEmail);
  }

  if (source === 'gmail' && userToken) {
    return new GmailAdapter(userToken);
  }

  // MCP fallback — only when no REST credentials available
  if (mcpServer?.enabled) {
    const toolName = SOURCE_TOOL_MAP[source];
    if (toolName) {
      return new MCPAdapter(mcpServer, toolName);
    }
  }

  return null; // No credentials configured — scanner will emit SSE warning
}
```

```typescript
// GET — return credentials with secrets masked (show last 4 chars only)
// POST — save credentials to settings.json under source_credentials key

import { NextRequest } from 'next/server';
import { readSettings, writeSettings, type SourceCredentials } from '@/lib/settings-core';

function maskSecret(value: string | undefined): string | undefined {
  if (!value) return undefined;
  return '****' + value.slice(-4);
}

export async function GET(): Promise<Response> {
  const settings = await readSettings();
  const creds = settings.source_credentials ?? {};

  // Mask all token/key values before returning to client
  const masked: Record<string, unknown> = {};
  if (creds.slack?.token) {
    masked.slack = { connected: true, tokenHint: maskSecret(creds.slack.token), channels: creds.slack.channels ?? [] };
  }
  if (creds.gong?.accessKey) {
    masked.gong = { connected: true, accessKeyHint: maskSecret(creds.gong.accessKey) };
  }
  if (creds.glean?.token) {
    masked.glean = { connected: true, tokenHint: maskSecret(creds.glean.token), instanceUrl: creds.glean.instanceUrl, actAsEmail: creds.glean.actAsEmail };
  }

  return Response.json(masked);
}

export async function POST(request: NextRequest): Promise<Response> {
  const body = await request.json() as Partial<SourceCredentials>;
  await writeSettings({ source_credentials: body });
  return Response.json({ ok: true });
}
```
