```typescript
import type { SourceAdapter } from './index';

export class SlackAdapter implements SourceAdapter {
  constructor(
    private readonly token: string,
    private readonly channels: string[]
  ) {}

  async fetchContent(query: string, since: string): Promise<string> {
    if (this.channels.length === 0) {
      console.warn('[SlackAdapter] No channels configured — skipping. Add channel IDs to Settings > Source Connections > Slack.');
      return '';
    }

    const oldest = String(new Date(since).getTime() / 1000); // ISO → Slack epoch seconds
    const parts: string[] = [];

    for (const channelId of this.channels) {
      try {
        const resp = await fetch('https://slack.com/api/conversations.history', {
          method: 'POST',
          headers: {
            'Authorization': `Bearer ${this.token}`,
            'Content-Type': 'application/json',
          },
          body: JSON.stringify({ channel: channelId, oldest, limit: 100 }),
        });

        const data = await resp.json() as { ok: boolean; messages?: Array<{ text?: string }> };

        if (!data.ok) {
          console.error(`[SlackAdapter] channel ${channelId}: Slack API error`);
          continue;
        }

        const messages = data.messages ?? [];
        for (const msg of messages) {
          if (msg.text) {
            parts.push(`[Slack #${channelId}] ${msg.text}`);
          }
        }
      } catch (err) {
        console.error(`[SlackAdapter] channel ${channelId} fetch failed:`, err instanceof Error ? err.message : err);
      }
    }

    return parts.join('\n');
  }
}
```

```typescript
import type { SourceAdapter } from './index';

interface GongCall { id: string; title?: string }
interface GongTranscriptSegment { speakerName?: string; sentences?: Array<{ text: string }> }

export class GongAdapter implements SourceAdapter {
  private readonly basicAuth: string;

  constructor(
    private readonly accessKey: string,
    private readonly accessKeySecret: string,
    private readonly baseUrl: string
  ) {
    this.basicAuth = Buffer.from(`${accessKey}:${accessKeySecret}`).toString('base64');
  }

  async fetchContent(query: string, since: string): Promise<string> {
    try {
      // Step 1: List calls in date range (with cursor pagination, max 3 pages)
      const callIds: string[] = [];
      let cursor: string | undefined;
      let page = 0;
      do {
        const url = new URL(`${this.baseUrl}/v2/calls`);
        url.searchParams.set('fromDateTime', since);
        if (cursor) url.searchParams.set('cursor', cursor);

        const listResp = await fetch(url.toString(), {
          headers: { 'Authorization': `Basic ${this.basicAuth}` },
        });
        const listData = await listResp.json() as { calls?: GongCall[]; cursor?: string };
        (listData.calls ?? []).forEach(c => callIds.push(c.id));
        cursor = listData.cursor;
        page++;
      } while (cursor && page < 3);

      if (callIds.length === 0) return '';

      // Step 2: Fetch transcripts for those call IDs
      const transcriptResp = await fetch(`${this.baseUrl}/v2/calls/transcript`, {
        method: 'POST',
        headers: {
          'Authorization': `Basic ${this.basicAuth}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ filter: { callIds } }),
      });

      const transcriptData = await transcriptResp.json() as {
        callTranscripts?: Array<{
          callId: string;
          transcript?: Array<{
            speakerName?: string;
            topic?: string;
            sentences?: Array<{ text: string }>;
          }>;
        }>;
      };

      const parts: string[] = [];
      for (const ct of transcriptData.callTranscripts ?? []) {
        for (const seg of ct.transcript ?? []) {
          const text = (seg.sentences ?? []).map(s => s.text).join(' ');
          if (text) {
            parts.push(`[Gong Call: ${ct.callId}] ${seg.speakerName ?? 'Unknown'}: ${text}`);
          }
        }
      }

      return parts.join('\n');
    } catch (err) {
      console.error('[GongAdapter] fetch failed:', err instanceof Error ? err.message : err);
      return '';
    }
  }
}
```
