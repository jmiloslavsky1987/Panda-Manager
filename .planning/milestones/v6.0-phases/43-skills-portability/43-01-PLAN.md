```typescript
// This function must be MOVED to lib/skill-path.ts, then re-exported here for backward compat
export function resolveSkillsDir(skillPath: string, dirnameRef: string = __dirname): string {
  const trimmed = skillPath.trim();
  if (!trimmed) {
    return path.join(dirnameRef, '../../skills');  // NOTE: fallback uses __dirname by default
  }
  if (trimmed.startsWith('/')) {
    return trimmed;
  }
  return path.join(os.homedir(), trimmed);
}
```
