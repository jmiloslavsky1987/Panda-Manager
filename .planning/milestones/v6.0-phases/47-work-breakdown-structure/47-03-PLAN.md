- Do NOT propose items that already exist (listed in the context).

  // 4. Build user message with project context + existing WBS item names
  const userMessage = `... [project data] ...\n\nExisting WBS items:\n${existingItems.map(i => `[${i.track} L${i.level}] ${i.name}`).join('\n')}`

  // 5. Call Anthropic SDK (claude-3-5-sonnet — established pattern)
  // Parse JSON response

  // 6. Validate and resolve parent_id:
  const proposals = parsed.filter(p => {
    if (p.level !== 2 && p.level !== 3) return false  // enforce L2/L3 only
    if (existingNames.has(p.name.toLowerCase().trim())) return false  // dedup

`handleConfirm`:
```typescript
async function handleConfirm() {
  setLoading(true)
  try {
    const parentIds: number[] = []
    for (const proposal of proposals) {
      const res = await fetch(`/api/projects/${projectId}/wbs`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({
          name: proposal.name,
          parent_id: proposal.parent_id,
          level: proposal.level,
          track: proposal.track,
        }),
      })
      if (!res.ok) throw new Error('Failed to save proposal')
      parentIds.push(proposal.parent_id)
    }
    setOpen(false)
    setProposals([])
    onConfirmed([...new Set(parentIds)])  // deduplicated parent IDs to auto-expand
    toast.success(`${proposals.length} tasks added to WBS`)
  } catch {
    toast.error('Failed to save some tasks — please try again')
  } finally {
    setLoading(false)
  }
}
```

Dialog footer: `<Button onClick={handleConfirm} disabled={loading || proposals.length === 0}>Add {proposals.length} tasks</Button>` and `<Button variant="outline" onClick={() => setOpen(false)}>Cancel</Button>`.
